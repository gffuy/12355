JavaScript 基础练习

练习目标

  • 变量定义：var、let、const
  • 变量类型：字符串、数字、布尔值、数组、对象
  • 表达式与算术运算
  • 流程控制：if、switch、for
  • 函数定义与调用

代码演示

// 变量定义
var studentName = '小明';
let studentAge = 16;
const className = '高一一班';

// 数据类型
const score = 88;
const passed = score >= 60;
const subjects = ['语文', '数学', '英语'];
const student = {
  name: studentName,
  age: studentAge,
  class: className,
  score: score,
  passed: passed,
};

// 函数定义
function getGrade(score) {
  if (score >= 90) {
    return 'A';
  } else if (score >= 80) {
    return 'B';
  } else if (score >= 70) {
    return 'C';
  } else if (score >= 60) {
    return 'D';
  }
  return 'F';
}

function sumNumbers(a, b, c) {
  return a + b + c;
}

// 控制流程
let message;
switch (getGrade(score)) {
  case 'A':
    message = '成绩优秀';
    break;
  case 'B':
  case 'C':
    message = '成绩良好';
    break;
  default:
    message = '需要继续努力';
}

// 循环计算 1 到 5 的和
let total = 0;
for (let i = 1; i <= 5; i++) {
  total += i;
}
